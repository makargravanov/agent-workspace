  component "Web UI\n(React + TypeScript)" as UI

package "Бизнес-логика" {
  component "Application Core" as AppCore

  component "Notification Service" as Notify
}

  component "Repository Layer" as RepoLayer
  database "PostgreSQL" as Db
}

interface "IApplicationFacade" as IApplicationFacade
interface "INotificationPort" as INotificationPort
interface "IRepositoryPort" as IRepositoryPort

UI --> IApplicationFacade : использует
IApplicationFacade - AppCore

AppCore --> Workflow : координирует
AppCore --> INotificationPort : отправляет события
AppCore --> IRepositoryPort : читает и изменяет данные

INotificationPort - Notify
IRepositoryPort - RepoLayer

RepoLayer --> Db : SQL
Notify --> AppCore : статус доставки

note right of AppCore
Управляет прикладными сценариями:
- задачи
- документы
- заметки
- рабочий контекст
end note

note right of RepoLayer
Инкапсулирует доступ
к сущностям предметной области
и поисковым индексам
end note
