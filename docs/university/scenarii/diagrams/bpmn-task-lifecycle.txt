@startuml
skinparam backgroundColor white
skinparam shadowing false
skinparam defaultFontName Times New Roman
skinparam activity {
  BackgroundColor white
  BorderColor black
  ArrowColor black
  DiamondBackgroundColor white
  DiamondBorderColor black
  BarColor black
  StartColor white
  EndColor white
  FontName Times New Roman
}

title BPMN-диаграмма процесса обработки задачи

|Координатор|
start
:Сформировать описание задачи;
:Инициировать создание задачи;

|Система «Контур»|
:Принять запрос на создание;
:Проверить обязательные поля и сроки;

if (Данные корректны?) then ([да])
  :Сохранить задачу;
  :Зафиксировать событие создания в аудите;
else ([нет])
  :Вернуть сообщение об ошибке заполнения;
  note right
  Исключительная ситуация:
  процесс завершается до записи
  задачи в базе данных.
  end note
  stop
endif

if (Есть блокирующие зависимости?) then ([да])
  :Установить статус «заблокирована»;
  :Промежуточное таймерное событие:\nожидание снятия блокировки;
  :Повторно проверить зависимости;
  if (Блокировка снята?) then ([да])
  else ([нет])
    :Отправить координатору сообщение\nо пересмотре плана;
    note right
    Исключительная ситуация:
    задача не может перейти
    в исполнение до снятия блокировки.
    end note
    stop
  endif
else ([нет])
endif

fork
  :Отправить уведомление исполнителю;
fork again
  :Запустить контроль срока\nи напоминания;
end fork

note right
Параллельный шлюз запускает одновременно
уведомление исполнителю и контроль сроков.
end note

repeat
  |Исполнитель|
  :Принять задачу в работу;
  :Выполнить работу по задаче;
  :Передать результат на проверку;

  |Система «Контур»|
  :Зафиксировать передачу результата;
  :Обновить статус задачи;

  |Координатор|
  :Проверить результат;
  if (Требуется доработка?) then ([да])
    |Система «Контур»|
    :Отправить замечания исполнителю;
    if (Срок выполнения просрочен?) then ([да])
      :Промежуточное таймерное событие:\nотправка напоминания;
    else ([нет])
    endif
  else ([нет])
  endif
repeat while (Требуется доработка?) is ([да])

|Система «Контур»|
:Перевести задачу в состояние «завершена»;
:Записать итоговое событие аудита;

stop
@enduml
