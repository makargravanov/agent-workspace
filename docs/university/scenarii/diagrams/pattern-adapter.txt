class AssetService {
  +uploadAsset(file: UploadFile) : Asset
  +downloadAsset(assetId: UUID) : BinaryAsset
}

interface AssetStorage {
  +put(file: BinaryAsset) : StorageKey
  +get(key: StorageKey) : BinaryAsset
  +delete(key: StorageKey)
}

class S3AssetStorageAdapter {
  -client : S3Client
}

class LocalAssetStorageAdapter {
  -fs : LocalFileSystem
}

class S3Client
class LocalFileSystem

class Asset {
  +id : UUID
  +fileName : String
  +storageKey : String
}

AssetService --> AssetStorage : использует
AssetStorage <|.. S3AssetStorageAdapter : реализует
AssetStorage <|.. LocalAssetStorageAdapter : реализует
S3AssetStorageAdapter --> S3Client : адаптирует API
LocalAssetStorageAdapter --> LocalFileSystem : адаптирует API
AssetService ..> Asset : управляет
